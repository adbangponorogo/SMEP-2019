SELECT
	id,
	id_skpd,
	nama,
	LCASE(username) username,
	password,
	status,
	email,
	telepon,
	alamat,
	tanggal_buat
FROM tb_users
WHERE id<>''
UNION ALL
SELECT
	a.id,
	IFNULL(b.id_skpd,'-') id_skpd,
	a.nama,
	LCASE(a.user) username,
	a.pass password,
	3 AS status,
	a.email,
	a.telepon,
	a.alamat,
	a.tanggal_buat
FROM spse_pegawai a
LEFT JOIN tb_users b ON a.user=b.username
